1、使用类成员时，前面加不加self.有什么区别

{% highlight ruby %}

//.h
@property (nonatomic, retain) NSString *name
 
//.m
name = @"bang"  //没有retain，随时会被释放
NSString *str = self.name  //等于NSString *str = [self name];
self.name = @"bang" //等于[self setName:@"bang"]; 这时在set方法里retain了这个字符串

{% endhighlight %}
